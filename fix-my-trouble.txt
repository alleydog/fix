# fix.sh
sudo nano /etc/systemd/system/fix.sh

# >> fix.sh
#!/bin/sh -e
sudo setkeycodes 3a 42
sudo iptables -t mangle -A POSTROUTING -j TTL --ttl-set 65

# fix.service
sudo nano /etc/systemd/system/fix.service

# >> fix.service
[Unit]
Description=Fix My Trouble
After=multi-user.target

[Service]
Type=idle
ExecStart=/etc/systemd/system/fix.sh

[Install]
WantedBy=multi-user.target

# chmod
sudo chmod +x /etc/systemd/system/fix.sh
sudo chmod 644 /etc/systemd/system/fix.service

# systemctl
sudo systemctl daemon-reload
sudo systemctl enable /etc/systemd/system/fix.service
