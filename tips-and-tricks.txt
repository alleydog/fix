powershell:
foreach ($i in Get-ChildItem .\*.wav) {echo "file '$i'" >> mylist.txt}

bat:
(for %%i in (*.wav) do @echo file '%%i') > mylist.txt

CLI:
for %i in (*.wav) do @echo file '%i') > mylist.txt

ffmpeg:
ffmpeg -f concat -safe 0 -i mylist.txt -c copy output.wav

extract:

convert:

cut:
